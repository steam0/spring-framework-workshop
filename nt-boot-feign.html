nt-boot-feign

Declarative HTTP clients — define an interface, configure a URL, and nt-boot handles the rest. Tracing headers are forwarded automatically.

<dependency>
    <groupId>no.norsktipping.boot</groupId>
    <artifactId>nt-boot-feign</artifactId>
</dependency>

1. Enable scanning

@FeignServiceScan
@SpringBootApplication
class Application
@FeignServiceScan
@SpringBootApplication
public class Application {
    public static void main(String[] args) {
        SpringApplication.run(Application.class, args);
    }
}

2. Define your client

@FeignService(key = "HelloWorld")
interface HelloWorldClient {

    @RequestLine("GET /hello")
    fun getHelloWorld(): HelloWorldResponse
}
@FeignService(key = "HelloWorld")
public interface HelloWorldClient {

    @RequestLine("GET /hello")
    HelloWorldResponse getHelloWorld();
}

3. Configure

nt:
  feign:
    clients:
      HelloWorld:
        url: "http://my-service:8080"
        readTimeout: 10000    # default
        connectTimeout: 3000  # default

4. Inject and use

@Service
class MyService(
    private val helloWorldClient: HelloWorldClient
) {
    fun doStuff(): String {
        val response = helloWorldClient.getHelloWorld()
        return response.message
    }
}
@Service
public class MyService {

    private final HelloWorldClient helloWorldClient;

    public MyService(HelloWorldClient helloWorldClient) {
        this.helloWorldClient = helloWorldClient;
    }

    public String doStuff() {
        HelloWorldResponse response = helloWorldClient.getHelloWorld();
        return response.getMessage();
    }
}
